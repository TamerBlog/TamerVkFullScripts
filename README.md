# Установка и запуск
1. git clone https://github.com/TamerBlog/TamerVkFullScripts
2. cd TamerVkFullScripts
3. sh install.sh
4. python start.py

# Настройка
1. cd Tools
2. nano config.py
3. Вставляем указанные параметры
